@override
void writeBytes(List<int> bytes, [int len]) {
  len ??= bytes.length;

  while (length + len > _buffer.length) {
    _expandBuffer((length + len) - _buffer.length);
  }
  _buffer.setRange(length, length + len, bytes);
  length += len;
}
